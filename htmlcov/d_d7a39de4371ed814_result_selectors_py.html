14def make_result_selectors(factors, dynamic_axes): 

15 """ 

16 Creates result selectors for an Altair chart. 

17 

18 Parameters 

19 ---------- 

20 factors : dict 

21 The factors to create selectors for. 

22 dynamic_axes : dict 

23 The dynamic axes to create selectors for. 

24 

25 Returns 

26 ------- 

27 dict 

28 Keyword arguments to add selections to metric charts and chart objects to 

29 display the selectors. 

30 """ 

31 dynamic_axes = dict(list(dynamic_axes.items())[: len(factors)]) 

32 

33 axes = make_axis_selectors(factors, dynamic_axes) 

34 parameters = make_parameter_filters(factors, dynamic_axes) 

35 

36 # kwargs to pass to metric charts 

37 metric_kwargs = { 

38 axis: {"shorthand": shorthand, "title": None} 

39 for axis, shorthand in dynamic_axes.items() 

40 } 

41 metric_kwargs["transform"] = axes["transforms"] + parameters["transforms"] 

42 

43 return { 

44 "kwargs": metric_kwargs, 

45 "charts": format_selector_charts(axes["charts"], parameters["charts"]), 

46 }

50 """

58 The dynamic axes to create selectors for. 

59 

60 Returns 

61 ------- 

62 dict 

63 The axis selectors and the charts to display them. 

64 """ 

65 factor_names = list(factors.keys()) 

66 

67 charts = [] 

68 transforms = [] 

69 for idx, axis in enumerate(dynamic_axes):

79 """ 

80 Creates parameter filters. 

81 

82 Parameters 

83 ---------- 

84 factors : dict 

85 The factors to create selectors for. 

86 dynamic_axes : dict 

87 The dynamic axes for a chart page. Used to determine initial selections. 

88 

89 Returns

91 dict 

92 The parameter filters and the charts to display them. 

93 """ 

94 n_dynamic_axes = len(dynamic_axes)

96 charts = []

98 for i, factor in enumerate(factors.items()):
