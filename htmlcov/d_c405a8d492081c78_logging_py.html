101@contextmanager 

102def multiprocessing_logging_queue(): 

103 """ 

104 Context manager for a logging queue that can be shared 

105 across multiple processes. 

106 """ 

107 with mp.Manager() as manager: 

108 try: 

109 logging_queue = manager.Queue() 

110 root_logger = get_logger("") 

111 listener = QueueListener(logging_queue, *root_logger.handlers) 

112 listener.start() 

113 yield logging_queue 

114 finally: 

115 listener.stop()
