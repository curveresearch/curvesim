14 def compute_trades(self, prices): # pylint: disable=arguments-differ 

15 """ 

16 Compute trades to arbitrage the pool, as follows: 

17 1. For each coin pair i and j, calculate size of coin i 

18 needed to move price of coin i w.r.t. to j to the 

19 target price. 

20 2. Calculate the profit from each such swap. 

21 3. Take the swap that gives the largest profit. 

22 

23 Parameters 

24 ---------- 

25 prices : pandas.Series 

26 Current market prices from the price_sampler. 

27 

28 Returns 

29 ------- 

30 trades : list of :class:`Trade` objects 

31 List of trades to perform. 

32 

33 additional_data: dict 

34 Dict of additional data to be passed to the state log as part of trade_data. 

35 """ 

36 pool = self.pool 

37 trades = get_arb_trades(pool, prices) 

38 

39 max_profit = 0 

40 best_trade = None 

41 price_error = None 

42 for t in trades: 

43 size, coins, price_target = t 

44 i, j = coins 

45 with pool.use_snapshot_context(): 

46 out_amount, _ = pool.trade(i, j, size) 

47 # assume we transacted at "infinite" depth at target price 

48 # on the other exchange to obtain our in-token 

49 profit = out_amount - size * price_target 

50 if profit > max_profit: 

51 max_profit = profit 

52 best_trade = Trade(i, j, size) 

53 price_error = pool.price(i, j) - price_target 

54 

55 if not best_trade: 

56 return [], {"price_errors": []} 

57 

58 return [best_trade], {"price_errors": [price_error]}
