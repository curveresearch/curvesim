13def from_address(address, chain, env="prod", end_ts=None): 

14 """ 

15 Returns 

16 

17 Parameters 

18 ---------- 

19 address: str 

20 Address prefixed with '0x' 

21 chain: str 

22 Chain name 

23 env: str 

24 Environment name for subgraph: 'prod' or 'staging' 

25 

26 Returns 

27 ------- 

28 Pool snapshot dictionary in the format returned by 

29 :func:`curvesim.network.subgraph.pool_snapshot`. 

30 """ 

31 loop = get_event_loop() 

32 data = pool_snapshot_sync(address, chain, env=env, end_ts=end_ts, event_loop=loop) 

33 

34 # Get underlying token addresses 

35 if data["pool_type"] == "LENDING": 

36 u_addrs, u_decimals = underlying_coin_info_sync( 

37 data["coins"]["addresses"], event_loop=loop 

38 ) 

39 

40 m = data.pop("coins") 

41 names = [n[1:] for n in m["names"]] 

42 

43 data["coins"] = { 

44 "names": names, 

45 "addresses": u_addrs, 

46 "decimals": u_decimals, 

47 "wrapper": m, 

48 } 

49 

50 return data
