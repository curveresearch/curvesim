13def make_chart_kwargs(default, override_dicts): 

14 """ 

15 Create the keyword arguments for a chart by updating a default dictionary with 

16 values from a list of override dictionaries. 

17 

18 Parameters 

19 ---------- 

20 default : dict 

21 The default keyword arguments for the chart. 

22 override_dicts : list 

23 A list of dictionaries containing addtional keyword arguments and/or overrides 

24 for the default keyword arguments. 

25 

26 Returns 

27 ------- 

28 dict 

29 A dictionary containing the keyword arguments for the chart. 

30 """ 

31 chart_kwargs = deepcopy(default) 

32 

33 for d in override_dicts: 

34 for key, val in d.items(): 

35 update = UPDATE_RULES.get(key, update_properties) 

36 update(chart_kwargs, key, val) 

37 

38 return init_properties(chart_kwargs)

41def update_properties(prop_dict, key, val, depth=0, count=0): 

42 """ 

43 Update properties in a dictionary. If the current depth matches the target depth, 

44 the value is updated directly. Otherwise, the function is called recursively on 

45 nested dictionaries. 

46 

47 Parameters 

48 ---------- 

49 prop_dict : dict 

50 The dictionary of properties to update. 

51 key : str 

52 The key of the property to update. 

53 val : any 

54 The new value for the property. 

55 depth : int, optional 

56 The target depth for the update. Defaults to 0. 

57 count : int, optional 

58 The current depth of the update. Defaults to 0. 

59 """ 

60 

61 if depth in [0, count]: 

62 prop_dict[key] = val 

63 else: 

64 prop_dict.setdefault(key, {}) 

65 for k, v in val.items(): 

66 update_properties(prop_dict[key], k, v, depth, count + 1)

69def concat_properties(prop_dict, key, val): 

70 """ 

71 Concatenate values to a property in a dictionary. If the value is a list, extend 

72 the property with it. Otherwise, append the value to the property. 

73 

74 Parameters 

75 ---------- 

76 prop_dict : dict 

77 The dictionary of properties to update. 

78 key : str 

79 The key of the property to update. 

80 val : any 

81 The value to concatenate to the property. 

82 """ 

83 prop_dict.setdefault(key, []) 

84 

85 if isinstance(val, list): 

86 prop_dict[key] += val 

87 else: 

88 prop_dict[key].append(val)

108def init_properties(prop_dict, classes=None): 

109 """ 

110 Initialize the properties of a dictionary using a mapping of classes. If a 

111 property's key is present in the classes dictionary, its value is replaced 

112 with an instance of the corresponding class, initialized with the value as 

113 keyword arguments. 

114 

115 Parameters 

116 ---------- 

117 prop_dict : dict 

118 The dictionary of properties to initialize. 

119 classes : dict, optional 

120 A dictionary mapping keys to classes. If a key in prop_dict matches a 

121 key in classes, the value in prop_dict is replaced with an instance of 

122 the corresponding class. Defaults to PROPERTY_CLASSES. 

123 

124 Returns 

125 ------- 

126 dict 

127 The dictionary of properties, with values replaced with class instances 

128 where applicable. 

129 """ 

130 classes = classes or PROPERTY_CLASSES 

131 for key, val in prop_dict.items(): 

132 try: 

133 prop_class = classes[key] 

134 except KeyError: 

135 continue 

136 

137 if isinstance(prop_class, dict): 

138 prop_dict[key] = init_properties(val, classes=classes[key]) 

139 else: 

140 prop_dict[key] = prop_class(**val) 

141 

142 return prop_dict
